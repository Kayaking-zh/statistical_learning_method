class Perceptron:
    '''
    
    '''
    def __init__(self, dim, eta):
        '''
        dim: 
        eta: 
        '''
        self.w = np.zeros(dim)
        self.eta = eta
        self.b = 0
    
    def fit(self, X, Y):
        wrong = True
        while wrong:
            wrong = False
            for x, y in zip(X, Y):
                if y * (np.dot(self.w, x) + self.b) <= 0:
                    self.w += self.eta * x * y
                    self.b += self.eta * y
                    wrong = True
